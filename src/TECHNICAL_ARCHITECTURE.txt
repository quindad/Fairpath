Pseudocode:
```
function checkEligibility(charges, probationCompleted, pendingCharges) {
  // Check disqualifying factors
  if (pendingCharges) return notEligible("Pending charges must be resolved first");
  if (!probationCompleted) return notEligible("Complete probation first");
  
  // Check state laws
  for each charge {
    law = getStateLaw(charge.state, charge.offense);
    
    if (law.neverEligible) return notEligible(law.reason);
    
    timeSince = currentYear - charge.year;
    if (timeSince < law.waitingPeriod) {
      return notEligible("Must wait " + (law.waitingPeriod - timeSince) + " more years");
    }
  }
  
  // If we got here, eligible!
  return eligible({
    reason: buildEligibilityReason(charges),
    nextSteps: getNextSteps(charges[0].state),
    costs: getCosts(charges[0].state),
    timeline: getTimeline(charges[0].state)
  });
}
```
